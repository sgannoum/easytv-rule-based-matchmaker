@prefix easyTV: <http://www.owl-ontologies.com/OntologyEasyTV.owl#> .
@prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns> .


[ ApplyAdaptationOnContent:	
	(?service rdf:#type ?type)
	(?type http://www.w3.org/2000/01/rdf-schema#subClassOf easyTV:ContentAccessibilityService)
	(?service easyTV:applies ?somthing)
	(?service easyTV:onContent ?object)
	(?object easyTV:contains ?otherObject)
	noValue(?service easyTV:onContent ?otherObject)
	->  	
	(?service easyTV:onContent ?otherObject)
]

[ SubstitutionCreation1:	
	(?service1 rdf:#type ?type1)
	(?type1 http://www.w3.org/2000/01/rdf-schema#subClassOf easyTV:ContentAccessibilityService)
	(?service1 easyTV:applies ?somthing)
	(?service1 easyTV:onContent ?object)
	(?service2 rdf:#type ?type2)
	(?type2 http://www.w3.org/2000/01/rdf-schema#subClassOf easyTV:ContentAccessibilityService)
	(?service2 easyTV:applies ?somthing)
	(?service2 easyTV:onContent ?object)
	strConcat(?service1, ?service1t)
	strConcat(?service2, ?service2t)
	NE(?service1t, ?service2t, ?res)
	noValue(?object easyTV:contains)
	noValue(?service1 easyTV:substitutePartOf ?service2)
	makeTemp(?sub)
	->  	
	(?service1 easyTV:substitutePartOf ?service2)
	(?sub rdf:#type easyTV:Substitution)
	(?sub easyTV:ofContentAdaptationService ?service1)
	(?sub easyTV:byContentAdaptationService ?service2)
]

[ SubstitutionConnection:	
	(?sub1 rdf:#type easyTV:Substitution)
	(?sub2 rdf:#type easyTV:Substitution)
	strConcat(?sub1, ?sub1t)
	strConcat(?sub2, ?sub2t)
	NE(?sub1t, ?sub2t, ?res)
	(?sub1 easyTV:ofContentAdaptationService ?service)
	(?sub2 easyTV:ofContentAdaptationService ?service)
	(?sub1 easyTV:byContentAdaptationService ?service1)
	(?sub2 easyTV:byContentAdaptationService ?service2)
	strConcat(?service1, ?service1t)
	strConcat(?service2, ?service2t)
	NE(?service1t, ?service2t, ?res1)
	(?service1 easyTV:applies ?somthing1)
	(?service2 easyTV:applies ?somthing2)
	(?somthing1 easyTV:excecutedBefore ?somthing2)
	->  	
	(?sub1 easyTV:nextSubstitution ?sub2)
]
